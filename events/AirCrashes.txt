#########################################

	id = crash.1
	title = crash.1.t
	desc = crash.1.d
	picture = GFX_event_test
	
	fire_only_once = yes
	
	trigger = {
		original_tag = KEN
		date > 2000.1.30
		date < 2001.1.1
	}
	

#Start search and rescue operation!
	option = {
		name = crash.1.a
		
		news_event = {
			hours = 6
			id = news_crash.1
		}
}

}
